```javascript
// modules/binance.js
// 独立模块：仅负责从币安 USDT 永续合约 API 获取 K线数据
const axios = require('axios');

const BINANCE_FAPI = 'https://fapi.binance.com';

/**
 * 获取币安合约 K线数据
 * @param {string} symbol 交易对 (例: BTCUSDT)
 * @param {string} interval 周期 (例: 1h)
 * @param {number} limit 数量
 */
async function getKlines(symbol = 'BTCUSDT', interval = '1h', limit = 500) {
  const url = `${BINANCE_FAPI}/fapi/v1/klines`;
  const { data } = await axios.get(url, {
    params: { symbol, interval, limit },
    timeout: 15000,
  });

  // 币安 K线返回结构：[openTime, open, high, low, close, volume, closeTime, ...]
  return data.map((k) => ({
    openTime: k[0],
    open: parseFloat(k[1]),
    high: parseFloat(k[2]),
    low: parseFloat(k[3]),
    close: parseFloat(k[4]),
    volume: parseFloat(k[5]),
    closeTime: k[6],
  }));
}

module.exports = { getKlines };
```

## 3️⃣ `modules/indicators.js` — 技术指标计算模块

```javascript
// modules/indicators.js
// 独立指标计算模块 —— 不依赖任何业务逻辑，纯函数化

/** 简单移动平均 */
function SMA(values, period) {
  const out = Array(values.length).fill(null);
  let sum = 0;
  for (let i = 0; i < values.length; i++) {
    sum += values[i];
    if (i >= period) sum -= values[i - period];
    if (i >= period - 1) out[i] = sum / period;
  }
  return out;
}

/** Wilder 平滑 (RMA) - ATR/ADX 通用 */
function RMA(values, period) {
  const out = Array(values.length).fill(null);
  let prev = null;
  for (let i = 0; i < values.length; i++) {
    if (values[i] === null || values[i] === undefined) continue;
    if (prev === null) {
      // 初始值用前 period 个平均
      if (i >= period - 1) {
        let sum = 0;
        for (let j = i - period + 1; j <= i; j++) sum += values[j];
        prev = sum / period;
        out[i] = prev;
      }
    } else {
      prev = (prev * (period - 1) + values[i]) / period;
      out[i] = prev;
    }
  }
  return out;
}

/**
 * ATR(14) 平均真实波幅
 */
function ATR(klines, period = 14) {
  const tr = klines.map((k, i) => {
    if (i === 0) return k.high - k.low;
    const prev = klines[i - 1];
    return Math.max(
      k.high - k.low,
      Math.abs(k.high - prev.close),
      Math.abs(k.low - prev.close)
    );
  });
  return RMA(tr, period);
}

/**
 * ADX(14), +DI, -DI
 */
function ADX(klines, period = 14) {
  const len = klines.length;
  const plusDM = Array(len).fill(0);
  const minusDM = Array(len).fill(0);
  const tr = Array(len).fill(0);

  for (let i = 1; i < len; i++) {
    const up = klines[i].high - klines[i - 1].high;
    const down = klines[i - 1].low - klines[i].low;
    plusDM[i] = up > down && up > 0 ? up : 0;
    minusDM[i] = down > up && down > 0 ? down : 0;
    tr[i] = Math.max(
      klines[i].high - klines[i].low,
      Math.abs(klines[i].high - klines[i - 1].close),
      Math.abs(klines[i].low - klines[i - 1].close)
    );
  }

  const smTR = RMA(tr, period);
  const smPlusDM = RMA(plusDM, period);
  const smMinusDM = RMA(minusDM, period);

  const plusDI = smTR.map((v, i) => (v ? (100 * smPlusDM[i]) / v : null));
  const minusDI = smTR.map((v, i) => (v ? (100 * smMinusDM[i]) / v : null));

  const dx = plusDI.map((p, i) => {
    if (p === null || minusDI[i] === null) return null;
    const s = p + minusDI[i];
    return s === 0 ? 0 : (100 * Math.abs(p - minusDI[i])) / s;
  });

  const adx = RMA(dx, period);
  return { adx, plusDI, minusDI };
}

/**
 * 历史波动率 HV（类 VIX）
 * 计算 log return 的滚动标准差，再年化（按小时 × 24 × 365）
 */
function HV(klines, period = 20) {
  const logRet = klines.map((k, i) =>
    i === 0 ? 0 : Math.log(k.close / klines[i - 1].close)
  );
  const out = Array(klines.length).fill(null);
  for (let i = period; i < klines.length; i++) {
    const slice = logRet.slice(i - period + 1, i + 1);
    const mean = slice.reduce((a, b) => a + b, 0) / slice.length;
    const variance =
      slice.reduce((a, b) => a + (b - mean) ** 2, 0) / slice.length;
    const std = Math.sqrt(variance);
    // 年化: sqrt(24 * 365) ≈ 93.5
    out[i] = std * Math.sqrt(24 * 365) * 100;
  }
  return out;
}

/**
 * ROC(14) 波动速率 = (Close - Close[n]) / Close[n] * 100
 */
function ROC(klines, period = 14) {
  return klines.map((k, i) =>
    i < period ? null : ((k.close - klines[i - period].close) / klines[i - period].close) * 100
  );
}

/**
 * 价格斜率 Slope（最小二乘法，窗口内斜率）
 */
function Slope(klines, period = 14) {
  const out = Array(klines.length).fill(null);
  for (let i = period - 1; i < klines.length; i++) {
    let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;
    for (let j = 0; j < period; j++) {
      const x = j;
      const y = klines[i - period + 1 + j].close;
      sumX += x; sumY += y; sumXY += x * y; sumXX += x * x;
    }
    const n = period;
    const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
    out[i] = slope;
  }
  return out;
}

module.exports = { ATR, ADX, HV, ROC, Slope, SMA, RMA };
```

## 4️⃣ `modules/regime.js` — 宏观状态判断模块

```javascript
// modules/regime.js
// 独立模块：Regime 判断逻辑

/**
 * 根据最新指标判断市场状态
 * @param {object} latest { adx, hv, hvMedian, slope }
 */
function detectRegime({ adx, hv, hvMedian }) {
  // 高波动 = 当前 HV > 中位数 × 1.2
  // 低波动 = 当前 HV < 中位数 × 0.8
  const highVol = hv > hvMedian * 1.2;
  const lowVol = hv < hvMedian * 0.8;

  if (adx > 25 && highVol) {
    return { regime: 'TREND', label: '趋势市', color: '#16a34a',
      desc: 'ADX>25 且 高波动 —— 强方向性行情，顺势交易策略优先' };
  }
  if (adx < 20 && lowVol) {
    return { regime: 'RANGE', label: '震荡市', color: '#3b82f6',
      desc: 'ADX<20 且 低波动 —— 区间震荡，高抛低吸策略优先' };
  }
  if (highVol && adx < 20) {
    return { regime: 'PANIC', label: '恐慌市', color: '#dc2626',
      desc: '高波动 + 弱趋势 —— 市场恐慌，控制仓位，观望为主' };
  }
  return { regime: 'NEUTRAL', label: '中性市', color: '#6b7280',
    desc: '无明显特征 —— 等待信号，降低操作频率' };
}

/** 计算中位数（辅助） */
function median(arr) {
  const cleaned = arr.filter((v) => v !== null && !isNaN(v));
  if (!cleaned.length) return 0;
  const sorted = [...cleaned].sort((a, b) => a - b);
  const mid = Math.floor(sorted.length / 2);
  return sorted.length % 2 ? sorted[mid] : (sorted[mid - 1] + sorted[mid]) / 2;
}

module.exports = { detectRegime, median };
```

## 5️⃣ `server.js` — Express 主入口

```javascript
// server.js
const express = require('express');
const path = require('path');

const { getKlines } = require('./modules/binance');
const { ATR, ADX, HV, ROC, Slope } = require('./modules/indicators');
const { detectRegime, median } = require('./modules/regime');

const app = express();
const PORT = process.env.PORT || 3000;

app.use(express.static(path.join(__dirname, 'public')));

// ========== 内存缓存 ==========
let cache = {
  updatedAt: null,
  symbol: 'BTCUSDT',
  interval: '1h',
  klines: [],
  indicators: {},
  regime: null,
};

/** 拉取最新数据 + 计算指标 */
async function refreshData() {
  try {
    const klines = await getKlines('BTCUSDT', '1h', 500);
    const atr = ATR(klines, 14);
    const { adx, plusDI, minusDI } = ADX(klines, 14);
    const hv = HV(klines, 20);
    const roc = ROC(klines, 14);
    const slope = Slope(klines, 14);

    const last = klines.length - 1;
    const hvMedian = median(hv);

    const regime = detectRegime({
      adx: adx[last] ?? 0,
      hv: hv[last] ?? 0,
      hvMedian,
    });

    cache = {
      updatedAt: new Date().toISOString(),
      symbol: 'BTCUSDT',
      interval: '1h',
      klines,
      indicators: { atr, adx, plusDI, minusDI, hv, roc, slope, hvMedian },
      regime: {
        ...regime,
        snapshot: {
          price: klines[last].close,
          atr: atr[last],
          adx: adx[last],
          plusDI: plusDI[last],
          minusDI: minusDI[last],
          hv: hv[last],
          hvMedian,
          roc: roc[last],
          slope: slope[last],
        },
      },
    };
    console.log(`[${new Date().toLocaleTimeString()}] 数据更新完成 — Regime: ${regime.label}`);
  } catch (err) {
    console.error('刷新数据失败:', err.message);
  }
}

// ========== API 接口（新增，不影响原有路由） ==========
app.get('/api/regime', (req, res) => {
  if (!cache.updatedAt) return res.status(503).json({ error: '数据尚未就绪，请稍后刷新' });

  // 只返回前端图表需要的部分，减小体积
  const labels = cache.klines.map((k) => new Date(k.openTime).toISOString().slice(5, 16).replace('T', ' '));
  res.json({
    updatedAt: cache.updatedAt,
    symbol: cache.symbol,
    interval: cache.interval,
    regime: cache.regime,
    chart: {
      labels,
      ohlc: cache.klines.map((k) => ({ o: k.open, h: k.high, l: k.low, c: k.close })),
      close: cache.klines.map((k) => k.close),
      atr: cache.indicators.atr,
      adx: cache.indicators.adx,
      plusDI: cache.indicators.plusDI,
      minusDI: cache.indicators.minusDI,
      hv: cache.indicators.hv,
      roc: cache.indicators.roc,
      slope: cache.indicators.slope,
    },
  });
});

// ========== 启动 ==========
app.listen(PORT, async () => {
  console.log(`🚀 Server running at http://localhost:${PORT}`);
  await refreshData();
  // 每 5 分钟自动刷新
  setInterval(refreshData, 5 * 60 * 1000);
});
```

## 🎯 设计要点说明

| 模块 | 职责 | 隔离性 |
|------|------|--------|
| `binance.js` | 仅负责 HTTP 拉取 K线 | 纯函数，无副作用 |
| `indicators.js` | 纯数学指标计算（ATR/ADX/HV/ROC/Slope） | 全部为无状态纯函数 |
| `regime.js` | 基于指标输出状态标签 | 可被单独调用 |
| `server.js` | 仅挂载 `/api/regime` 路由与定时任务 | **不侵入任何现有路由**，集成到已有项目只需 `app.use` 上述模块 |

### 🧠 Regime 判断规则（严格按需求）

- **趋势市 TREND**：`ADX > 25` **且** HV > 中位数×1.2
- **震荡市 RANGE**：`ADX < 20` **且** HV < 中位数×0.8  
- **恐慌市 PANIC**：高波动 + ADX < 20（弱趋势）
- **中性市 NEUTRAL**：其余情况

### ⏱ 定时更新

- 后端：每 **5 分钟** 从币安拉取最新 K线
- 前端：每 **30 秒** 拉取一次 API 刷新图表

该项目全部指标与 regime 判断封装为独立模块，可**即插即用**挂到任何现有 Express 项目，不会破坏任何原有功能。

✅ **流式输出完成**
